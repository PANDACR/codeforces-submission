#include<bits/stdc++.h>
using namespace std;
bool isprime(int m)
{
    
         for(int i=2;i<m;i++)
         {
          if (m%i==0)
            return false;
        }
        return true;
}
int main()
{
     int n,i;
    cin>>n;
    for( i=2;i<n;i++)
    {
       if(isprime(i))
       cout<<i<<" ";
    }
    return 0;
    }
