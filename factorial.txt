#include<bits/stdc++.h>
using namespace std;

void fact(int m)
{
    long long int result=1;
    if(m==1&&m==0)
    {
        cout<<"1"<<endl;
    }
    else
    {
        for(int i=1;i<=m;i++)
        {
            result=result*i;
        }
        cout<<result<<endl;

    }

}
int main()
{
    int n,m;

    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>m;
        fact(m);


    }
}
